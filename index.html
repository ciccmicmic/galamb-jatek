function update() { // Parallax mozgás (sebességhez kötve) bgX -= gameSpeed * 0.15; // Lassított tempo if (isTouching && (Date.now() - touchStartTime > 200)) { player.isDucking = true; } if (player.isJumping) { player.y += player.velY; player.velY += player.gravity; if (player.y >= 400) { player.y = 400; player.isJumping = false; } } // --- AKADÁLY SPAWN - ARÁNYOSÍTVA --- obstacleTimer++; if (obstacleTimer > 130 && Math.random() < 0.01) { const isHeli = Math.random() > 0.7; // Kiszámoljuk az akadály ideális szélességét és magasságát // pl. a vászon magasságának 15%-a (0.15) const obsSize = canvas.height * 0.15; obstacles.push({ x: canvas.width, // A magassági pozíciót is arányosítjuk (fix koordináták helyett) y: isHeli ? canvas.height * 0.45 : canvas.height * 0.72, w: obsSize, // Arányos szélesség h: obsSize, // Arányos magasság img: isHeli ? imgHeli[0] : imgAvo[0] }); obstacleTimer = 0; } obstacles.forEach((obs, i) => { obs.x -= gameSpeed; // Ütközés finomhangolás (az új arányos mérethez) if (player.x < obs.x + obs.w - 20 && player.x + player.w - 20 > obs.x + 20 && (player.isDucking ? player.y + 50 : player.y) < obs.y + obs.h - 20 && player.y + player.h > obs.y + 20) { sndHit.play(); bgMusic.pause(); gameRunning = false; setTimeout(() => location.reload(), 1500); } if (obs.x < -100) obstacles.splice(i, 1); }); points++; if (points % 600 === 0) gameSpeed += 0.4; player.frameTimer++; } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // --- HÁTTÉR RAJZOLÁS (Marad a régi, arányos) --- const bgScale = canvas.height / imgBg.height; const bgScaledWidth = imgBg.width * bgScale; const drawX = Math.floor(bgX); ctx.drawImage(imgBg, drawX, 0, bgScaledWidth, canvas.height); ctx.drawImage(imgBg, drawX + Math.floor(bgScaledWidth), 0, bgScaledWidth, canvas.height); if (bgX <= -bgScaledWidth) bgX = 0; // --- JÁTÉKOS RAJZOLÁS (Fix koordináták, mert a vászon fix) --- let currentImg = imgRun[Math.floor(player.frameTimer / 6) % 2]; if (player.isJumping) currentImg = imgJump; if (player.isDucking) currentImg = imgDuck[Math.floor(player.frameTimer / 6) % 2]; const drawH = player.isDucking ? player.h * 0.7 : player.h; const drawY = player.isDucking ? player.y + (player.h * 0.3) : player.y; ctx.drawImage(currentImg, player.x, drawY, player.w, drawH); // --- AKADÁLYOK RAJZOLÁSA - ARÁNYOSÍTVA --- obstacles.forEach(obs => { // A drawImage most már a obs.w és obs.h arányos méreteket használja ctx.drawImage(obs.img, obs.x, obs.y, obs.w, obs.h); }); // --- SZÖVEG --- ctx.fillStyle = "black"; ctx.font = "bold 26px Arial"; ctx.fillText(`NER: ${playerName} | Közpénz: ${points}`, 20, 40); }
